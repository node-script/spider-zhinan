								function getFiles(policyid){
									var files = "";

									$.post("/getPolicyFile.jspx",{"policyid":policyid},function(data){

												files += "<span style='color:blue;cursor:pointer;' onclick=\"downloadFile('"+policy.fileid+"')\">"+policy.filename+"";
											}
										}else{
											files = "æ— ";
										}
										
									});
									$.ajaxSettings.async = true;
									return files;
								}
